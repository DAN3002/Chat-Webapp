- chatRooms:
	+ _id: ObjectID
	+ members: Array(userID)
	+ messages:
		* id
		* sender: userID
		* text: String
		* timestamp: Date
		* img: String
	+ name: String - optional
	+ lastMessage:
		* like messages
	+ type: Private or Group

- users:
	+ _id: ObjectID
	+ username: String
	+ fullName: String
	+ email: String
	+ phone: String
	+ role: User or Chatbot
